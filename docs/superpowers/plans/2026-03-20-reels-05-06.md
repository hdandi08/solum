<script>
const SCENES = [
  { id: 's1', duration: 2500 },
  { id: 's2', duration: 3000 },
  { id: 's3', duration: 7000 },
  { id: 's4', duration: 4000 },
  { id: 's5', duration: 4000 },
];

const TOTAL = SCENES.reduce((a, s) => a + s.duration, 0);

let currentScene = -1;
let timer = null;
let revealTimer = null;
let startTime = null;
let rafId = null;
let running = false;

function showScene(index) {
  if (currentScene >= 0 && SCENES[currentScene]) {
    const prev = document.getElementById(SCENES[currentScene].id);
    if (prev) prev.classList.remove('reveal-active');
  }

  SCENES.forEach(s => {
    const el = document.getElementById(s.id);
    el.classList.remove('active');
    el.querySelectorAll('.txt').forEach(t => t.classList.remove('visible'));
  });

  if (index >= SCENES.length) { running = false; return; }

  currentScene = index;
  const el = document.getElementById(SCENES[index].id);
  el.classList.add('active');

  el.querySelectorAll('.txt').forEach((t, i) => {
    const delay = parseInt(t.dataset.delay || 0) + i * 120;
    setTimeout(() => t.classList.add('visible'), delay);
  });

  if (SCENES[index].id === 's3') {
    const eyebrow = el.querySelector('.s3-eyebrow');
    if (eyebrow) setTimeout(() => eyebrow.classList.add('visible'), 150);
    const list = el.querySelector('.ingredient-list');
    if (list) list.classList.add('scroll-running');
    revealTimer = setTimeout(() => el.classList.add('reveal-active'), 3900);
  }

  timer = setTimeout(() => showScene(index + 1), SCENES[index].duration);
}

function updateProgress() {
  if (!running) return;
  const elapsed = Date.now() - startTime;
  const pct = Math.min((elapsed / TOTAL) * 100, 100);
  document.getElementById('progress-bar').style.width = pct + '%';
  if (pct < 100) rafId = requestAnimationFrame(updateProgress);
}

function startReel() {
  const overlay = document.getElementById('play-overlay');
  const icon = overlay.querySelector('.play-icon');
  const label = overlay.querySelector('.play-label');
  const dur = overlay.querySelector('.play-duration');
  icon.style.display = 'none';
  dur.style.display = 'none';
  label.style.fontSize = '18px';
  label.style.letterSpacing = '0.1em';
  let count = 10;
  label.textContent = 'STARTING IN ' + count + '…';
  const tick = setInterval(() => {
    count--;
    if (count > 0) {
      label.textContent = 'STARTING IN ' + count + '…';
    } else {
      clearInterval(tick);
      overlay.style.display = 'none';
      restartReel();
    }
  }, 1000);
}

function restartReel() {
  clearTimeout(timer);
  clearTimeout(revealTimer);
  cancelAnimationFrame(rafId);
  running = true;
  startTime = Date.now();
  document.getElementById('progress-bar').style.width = '0%';
  document.getElementById('s3').classList.remove('reveal-active');
  const eyebrow = document.querySelector('#s3 .s3-eyebrow');
  if (eyebrow) eyebrow.classList.remove('visible');
  const list = document.querySelector('#s3 .ingredient-list');
  if (list) list.classList.remove('scroll-running');
  showScene(0);
  rafId = requestAnimationFrame(updateProgress);
}

function toggleRecordMode() {
  document.body.classList.toggle('record-mode');
  const btn = document.querySelector('.btn-record');
  const stage = document.getElementById('stage');
  if (document.body.classList.contains('record-mode')) {
    btn.textContent = '✕ Exit Record Mode';
    document.getElementById('play-overlay').style.display = 'none';
    const scale = Math.min(window.innerHeight / 720, window.innerWidth / 405);
    stage.style.transform = `scale(${scale})`;
    startCountdown();
  } else {
    btn.textContent = '⬤ Record Mode';
    stage.style.transform = '';
    clearTimeout(timer);
    cancelAnimationFrame(rafId);
    document.getElementById('countdown-overlay').classList.remove('active');
  }
}

function startCountdown() {
  SCENES.forEach(s => {
    const el = document.getElementById(s.id);
    el.classList.remove('active');
    el.querySelectorAll('.txt').forEach(t => t.classList.remove('visible'));
  });
  document.getElementById('progress-bar').style.width = '0%';
  const overlay = document.getElementById('countdown-overlay');
  const num = document.getElementById('countdown-number');
  const label = document.getElementById('countdown-label');
  overlay.classList.add('active');
  let count = 30;
  num.textContent = count;
  label.textContent = 'Start QuickTime recording now';
  const tick = setInterval(() => {
    count--;
    if (count === 10) label.textContent = 'Hit Record now ↑';
    if (count > 0) {
      num.textContent = count;
    } else {
      clearInterval(tick);
      num.textContent = '▶';
      label.textContent = 'Recording…';
      setTimeout(() => { overlay.classList.remove('active'); restartReel(); }, 400);
    }
  }, 1000);
}
</script>

<script>
const SCENES = [
  { id: 's1', duration: 2500 },
  { id: 's2', duration: 3000 },
  { id: 's3', duration: 7000 },
  { id: 's4', duration: 4000 },
  { id: 's5', duration: 4000 },
];

const TOTAL = SCENES.reduce((a, s) => a + s.duration, 0);

let currentScene = -1;
let timer = null;
let countdownTimers = []; // Array — stores all setTimeout IDs for S3 countdown
let startTime = null;
let rafId = null;
let running = false;

function showScene(index) {
  if (currentScene >= 0 && SCENES[currentScene]) {
    const prev = document.getElementById(SCENES[currentScene].id);
    if (prev) prev.classList.remove('reveal-active');
  }

  SCENES.forEach(s => {
    const el = document.getElementById(s.id);
    el.classList.remove('active');
    el.querySelectorAll('.txt').forEach(t => t.classList.remove('visible'));
  });

  if (index >= SCENES.length) { running = false; return; }

  currentScene = index;
  const el = document.getElementById(SCENES[index].id);
  el.classList.add('active');

  el.querySelectorAll('.txt').forEach((t, i) => {
    const delay = parseInt(t.dataset.delay || 0) + i * 120;
    setTimeout(() => t.classList.add('visible'), delay);
  });

  // S3 countdown chain
  if (SCENES[index].id === 's3') {
    countdownTimers = [];
    const counterEl = document.getElementById('s3-countdown-num');
    const timerBlock = document.getElementById('s3-timer-block');
    const stat = document.getElementById('s3-stat');

    const nums = ['3', '2', '1', '0'];
    const delays = [0, 1000, 2000, 3200];

    nums.forEach((n, i) => {
      countdownTimers.push(setTimeout(() => {
        counterEl.textContent = n;
        if (n === '0') {
          counterEl.classList.add('pulse');
          countdownTimers.push(setTimeout(() => {
            counterEl.classList.remove('pulse');
            // Fade timer block out via inline style (overrides .txt.visible opacity:1)
            timerBlock.style.opacity = '0';
            timerBlock.style.transition = 'opacity 0.4s ease';
            stat.classList.add('stat-visible');
          }, 300));
        }
      }, delays[i]));
    });
  }

  timer = setTimeout(() => showScene(index + 1), SCENES[index].duration);
}

function updateProgress() {
  if (!running) return;
  const elapsed = Date.now() - startTime;
  const pct = Math.min((elapsed / TOTAL) * 100, 100);
  document.getElementById('progress-bar').style.width = pct + '%';
  if (pct < 100) rafId = requestAnimationFrame(updateProgress);
}

function startReel() {
  const overlay = document.getElementById('play-overlay');
  const icon = overlay.querySelector('.play-icon');
  const label = overlay.querySelector('.play-label');
  const dur = overlay.querySelector('.play-duration');
  icon.style.display = 'none';
  dur.style.display = 'none';
  label.style.fontSize = '18px';
  label.style.letterSpacing = '0.1em';
  let count = 10;
  label.textContent = 'STARTING IN ' + count + '…';
  const tick = setInterval(() => {
    count--;
    if (count > 0) {
      label.textContent = 'STARTING IN ' + count + '…';
    } else {
      clearInterval(tick);
      overlay.style.display = 'none';
      restartReel();
    }
  }, 1000);
}

function restartReel() {
  clearTimeout(timer);
  countdownTimers.forEach(id => clearTimeout(id));
  countdownTimers = [];
  cancelAnimationFrame(rafId);
  running = true;
  startTime = Date.now();
  document.getElementById('progress-bar').style.width = '0%';
  // Reset S3 countdown state
  const counterEl = document.getElementById('s3-countdown-num');
  counterEl.textContent = '3';
  counterEl.classList.remove('pulse');
  const timerBlock = document.getElementById('s3-timer-block');
  timerBlock.style.opacity = '';
  timerBlock.style.transition = '';
  document.getElementById('s3-stat').classList.remove('stat-visible');
  showScene(0);
  rafId = requestAnimationFrame(updateProgress);
}

function toggleRecordMode() {
  document.body.classList.toggle('record-mode');
  const btn = document.querySelector('.btn-record');
  const stage = document.getElementById('stage');
  if (document.body.classList.contains('record-mode')) {
    btn.textContent = '✕ Exit Record Mode';
    document.getElementById('play-overlay').style.display = 'none';
    const scale = Math.min(window.innerHeight / 720, window.innerWidth / 405);
    stage.style.transform = `scale(${scale})`;
    startCountdown();
  } else {
    btn.textContent = '⬤ Record Mode';
    stage.style.transform = '';
    clearTimeout(timer);
    cancelAnimationFrame(rafId);
    document.getElementById('countdown-overlay').classList.remove('active');
  }
}

function startCountdown() {
  SCENES.forEach(s => {
    const el = document.getElementById(s.id);
    el.classList.remove('active');
    el.querySelectorAll('.txt').forEach(t => t.classList.remove('visible'));
  });
  document.getElementById('progress-bar').style.width = '0%';
  const overlay = document.getElementById('countdown-overlay');
  const num = document.getElementById('countdown-number');
  const label = document.getElementById('countdown-label');
  overlay.classList.add('active');
  let count = 30;
  num.textContent = count;
  label.textContent = 'Start QuickTime recording now';
  const tick = setInterval(() => {
    count--;
    if (count === 10) label.textContent = 'Hit Record now ↑';
    if (count > 0) {
      num.textContent = count;
    } else {
      clearInterval(tick);
      num.textContent = '▶';
      label.textContent = 'Recording…';
      setTimeout(() => { overlay.classList.remove('active'); restartReel(); }, 400);
    }
  }, 1000);
}
